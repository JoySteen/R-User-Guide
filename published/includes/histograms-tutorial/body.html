plotly_iframe <- function(url) {
    # set width and height from options or default square
    w <- "600"
    h <- "600"
    html <- paste("<iframe height=\"", h, "\" id=\"igraph\" scrolling=\"no\" seamless=\"seamless\"\n\t\t\t\tsrc=\"", 
        url, "\" width=\"", w, "\" frameBorder=\"0\"></iframe>", sep="")
    return(html)
}
